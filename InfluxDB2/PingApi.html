def ping
  uri = _parse_uri('/ping')
  headers = _get(uri)
  Ping.new.tap do |model|
    model.status = 'ok'
    model.build = headers['X-Influxdb-Build']
    model.version = headers['X-Influxdb-Version']
    model.message = 'ready for queries and writes'
  end
rescue StandardError => e
  Ping.new.tap do |model|
    model.status = 'fail'
    model.message = e.message
  end
end
