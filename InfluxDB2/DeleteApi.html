def delete(start, stop, predicate: nil, bucket: nil, org: nil)
  delete_request = InfluxDB2::DeletePredicateRequest.new(start: _to_rfc3339(start), stop: _to_rfc3339(stop),
                                                         predicate: predicate)

  _delete(delete_request, bucket: bucket, org: org)
end
